import numpy 
import smtplib
li = [] 
# number of elemetns as input 
n = int(input("Enter number of elements : ")) 
# iterating till the range 
for i in range(0, n): 
    ele = str(input()) 
    li.append(ele)  
for dest in li: 
    s = smtplib.SMTP('smtp.gmail.com', 587) 
    s.starttls() 
    s.login("sender@gmail.com", "sender password") 
    message = "Message_you_need_to_send"
    s.sendmail("sender@gmail.com", dest, message) 
    s.quit()
